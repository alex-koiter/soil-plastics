Study period weather

Download

In [9]:
df_day <- weather_dl(station_ids = 29886, interval = "day", start = "2021-10-01", end = "2023-09-30") |>
  select(date, mean_temp, total_precip) |>
  mutate(water_year = case_when(
    month(date) >= 10 ~ year(date) + 1,
    TRUE ~ year(date)
  )) |>
  group_by(water_year) %>%
  mutate(cumul = cumsum(replace_na(total_precip, 0))) %>%
  ungroup()
